# Momentum Strategy v1.0 class MomentumStrategy: def __init__(self, lookback=20, threshold=0.02): self.lookback = lookback self.threshold = threshold def on_tick(self, price_history): # Simple rate-of-change momentum roc = (price_history[-1] - price_history[-self.lookback]) / price_history[-self.lookback] if roc > self.threshold: return Signal.BUY, abs(roc) elif roc < -self.threshold: return Signal.SELL, abs(roc) else: return Signal.HOLD, 0.0
